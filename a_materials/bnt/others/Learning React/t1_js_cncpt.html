JavaScript Destructuring – Beginner to Ultra Pro

1. Array Destructuring

Basic Example


const numbers = [10, 20, 30];
const [first, second, third] = numbers;
console.log(first);  // 10
console.log(second); // 20
console.log(third);  // 30

Skipping Elements


const data = [1, 2, 3, 4];
const [a, , c] = data;
console.log(a); // 1
console.log(c); // 3

Default Values


const arr = [5];
const [x, y = 10] = arr;
console.log(x); // 5
console.log(y); // 10

2. Object Destructuring

Basic Example


const person = { name: "Ali", age: 22 };
const { name, age } = person;
console.log(name); // "Ali"
console.log(age);  // 22

Renaming Variables


const user = { id: 101, role: "admin" };
const { id: userId, role: userRole } = user;
console.log(userId);   // 101
console.log(userRole); // "admin"

Default Values


const settings = { theme: "dark" };
const { theme, fontSize = 14 } = settings;
console.log(theme);    // "dark"
console.log(fontSize); // 14

3. Nested Destructuring

Nested Object


const student = {
  name: "Sara",
  marks: {
    maths: 90,
    physics: 85
  }
};
const {
  marks: { maths, physics }
} = student;
console.log(maths);  // 90
console.log(physics);// 85

Nested Array


const numbers = [1, [2, 3], 4];
const [a, [b, c], d] = numbers;
console.log(a, b, c, d); // 1 2 3 4

4. Function Parameter Destructuring

Object as Parameter


function introduce({ name, age }) {
  console.log(`My name is ${name} and I am ${age} years old.`);
}
const user = { name: "Ahmed", age: 25 };
introduce(user);

Array as Parameter


function sum([a, b]) {
  return a + b;
}
console.log(sum([5, 7])); // 12

5. Ultra Pro Level

Destructuring with Rest Operator


const fruits = ["apple", "banana", "mango", "orange"];
const [first, ...rest] = fruits;
console.log(first); // "apple"
console.log(rest);  // ["banana", "mango", "orange"]

Combined Object & Array Destructuring


const data = {
  title: "Developer",
  skills: ["JavaScript", "React", "Node.js"]
};
const {
  title,
  skills: [primary, secondary, tertiary]
} = data;
console.log(title);    // "Developer"
console.log(primary);  // "JavaScript"
console.log(secondary);// "React"
console.log(tertiary); // "Node.js"

Nested with Renaming and Default


const employee = {
  id: 501,
  profile: {
    name: "Bilal",
    dept: "IT"
  }
};
const {
  profile: { name: employeeName, dept = "General" },
  role = "Intern"
} = employee;
console.log(employeeName); // "Bilal"
console.log(dept);         // "IT"
console.log(role);         // "Intern"

Destructuring in Loops


const users = [
  { name: "Ali", age: 22 },
  { name: "Sara", age: 25 }
];
for (const { name, age } of users) {
  console.log(`${name} is ${age} years old.`);
}

Destructuring Returned Values


function getScores() {
  return [70, 80, 90];
}
const [maths, physics, chemistry] = getScores();
console.log(maths, physics, chemistry); // 70 80 90
